# Gerador de Contratos — Flask (Python) Migração do seu app web para **Python + Flask**, mantendo o **frontend Tailwind/JS** praticamente igual e movendo a geração de **DOCX → PDF nativo** para o backend em Python. A UI visual (Tailwind, ícones Lucide, toasts, modal de progresso, confetti) permanece. --- ## Como executar 1. **Salvar estrutura de pastas** (exemplo): ``` projeto/ ├─ app.py ├─ requirements.txt ├─ templates/ │ └─ index.html └─ static/ └─ (vazio ou assets próprios) └─ templates_docs/ ├─ InovaInox/ │ ├─ Contrato.docx │ ├─ Ordem_de_Servico.docx │ └─ Capa.docx └─ PortoStone/ ├─ Contrato.docx └─ Ordem_de_Servico.docx ``` 2. **Instalar dependências**: ``` python -m venv .venv source .venv/bin/activate # (Windows: .venv\Scripts\activate) pip install -r requirements.txt ``` 3. **Executar**: ``` python app.py ``` Acesse: [http://127.0.0.1:5000](http://127.0.0.1:5000) > Para gerar **PDF com texto selecionável**, no Windows/Mac o `docx2pdf` usa o Word instalado. No Linux, instale o LibreOffice e o app usa `soffice --headless` automaticamente. --- ## requirements.txt ``` Flask==3.0.3 python-docx==1.1.2 docxtpl==0.16.7 num2words==0.5.13 docx2pdf==0.1.8 ``` > No Linux, `docx2pdf` geralmente não funciona; usamos LibreOffice. Instale via pacote da distro (ex.: `sudo apt install libreoffice`). --- ## app.py (Flask backend — geração DOCX e PDF nativo) ```python import io import os import sys import json import base64 import zipfile import platform import tempfile import subprocess from pathlib import Path from datetime import datetime from flask import Flask, send_from_directory, send_file, request, jsonify from docxtpl import DocxTemplate from docx import Document from docx.shared import Mm from num2words import num2words from decimal import Decimal, ROUND_HALF_UP try: from docx2pdf import convert as docx2pdf_convert except Exception: docx2pdf_convert = None APP_ROOT = Path(__file__).parent.resolve() TEMPLATES_DIR = APP_ROOT / "templates_docs" # onde ficam os .docx originais BUILD_DIR = APP_ROOT / "_build" BUILD_DIR.mkdir(exist_ok=True) app = Flask(__name__, static_folder="static", template_folder="templates") # ===== Helpers ===== def brl_decimal(n): if isinstance(n, str): clean = n.replace(" ", "").replace(".", "").replace(",", ".") try: n = Decimal(clean) except Exception: n = Decimal("0") else: n = Decimal(str(n)) return n.quantize(Decimal("0.01"), rounding=ROUND_HALF_UP) def brl_fmt(n): n = brl_decimal(n) s = f"{n:,.2f}" # 12,345.67 s = s.replace(",", "X").replace(".", ",").replace("X", ".") return f"R$ {s}" def por_extenso_brl(n): n = brl_decimal(n) inteiro = int(n) cent = int((n - Decimal(inteiro)) * 100) base = num2words(inteiro, lang='pt_BR') if cent > 0: return f"{base} reais e {num2words(cent, lang='pt_BR')} centavos" return f"{base} reais" def ensure_parent(p: Path): p.parent.mkdir(parents=True, exist_ok=True) return p def gerar_pdf(caminho_docx: Path, caminho_pdf: Path): """Gera PDF com texto selecionável. - Windows/Mac: tenta usar docx2pdf (Word) - Linux: LibreOffice headless """ ensure_parent(caminho_pdf) system = platform.system() if system in ("Windows", "Darwin") and docx2pdf_convert is not None: # Word instalado docx2pdf_convert(str(caminho_docx), str(caminho_pdf)) if not caminho_pdf.exists(): raise RuntimeError("docx2pdf não gerou o PDF.") return # Linux (ou fallback): LibreOffice try: subprocess.check_call([ "soffice", "--headless", "--convert-to", "pdf", "--outdir", str(caminho_pdf.parent), str(caminho_docx) ]) except FileNotFoundError: raise RuntimeError( "LibreOffice (soffice) não encontrado. Instale-o para gerar PDF nativo." ) if not caminho_pdf.exists(): raise RuntimeError("LibreOffice não gerou o PDF.") def decode_images_b64(images_b64): arr = [] for item in images_b64 or []: try: # item: {name: str, data_url: 'data:image/png;base64,...'} data_url = item.get('data_url') if not data_url or 'base64,' not in data_url: continue b64 = data_url.split('base64,', 1)[1] arr.append(base64.b64decode(b64)) except Exception: pass return arr def anexar_imagens_no_final(docx_path: Path, images_bytes: list): if not images_bytes: return doc = Document(str(docx_path)) # nova página doc.add_page_break() for i, img in enumerate(images_bytes): tmp = tempfile.NamedTemporaryFile(delete=False, suffix=f"_{i}.png") tmp.write(img) tmp.flush() tmp.close() # largura ~160 mm (A4 - margens), ajuste conforme desejar doc.add_picture(tmp.name, width=Mm(160)) os.unlink(tmp.name) doc.save(str(docx_path)) def normalize_context_keys(data: dict) -> dict: """Cria aliases ASCII para chaves com acento (ex.: NÚMERO_PARCELAS -> NUMERO_PARCELAS). O docxtpl/Jinja lida melhor com chaves ASCII. Mantemos ambas para cobrir templates. """ import unicodedata ctx = dict(data) for k, v in list(data.items()): norm = unicodedata.normalize('NFD', k) norm = ''.join(ch for ch in norm if unicodedata.category(ch) != 'Mn') if norm != k and norm not in ctx: ctx[norm] = v return ctx # ===== Rotas ===== @app.route('/') def index(): return send_from_directory(app.template_folder, 'index.html') @app.route('/api/generate', methods=['POST']) def api_generate(): """Recebe JSON: {company, docs, data, images} - company: 'InovaInox' | 'PortoStone' - docs: ['Contrato', 'Ordem_de_Servico', 'Capa'] - data: dict de chaves -> valores - images: [{name, data_url}, ...] (opcional) Retorna: - único PDF (se 1 doc) ou ZIP (se >1) """ try: payload = request.get_json(force=True) except Exception: return jsonify({"error": "JSON inválido"}), 400 company = (payload or {}).get('company') docs = (payload or {}).get('docs') or [] data = (payload or {}).get('data') or {} images_b64 = (payload or {}).get('images') or [] if not company: return jsonify({"error": "company ausente"}), 400 if not docs: return jsonify({"error": "docs vazio"}), 400 # Calculados financeiros (igual ao front, mas garantimos no backend) v_prod = brl_decimal(data.get('VALOR_PRODUTOS_INSTALACAO', 0)) v_desc = brl_decimal(data.get('VALOR_DESCONTO', 0)) total = (v_prod - v_desc) data['VALOR_TOTAL'] = brl_fmt(total) data['VALOR_TOTAL_POR_EXTENSO'] = por_extenso_brl(total) entrada_flag = str(data.get('VALOR_ENTRADA') or '').strip() != '' v_ent = brl_decimal(data.get('VALOR_ENTRADA', 0)) if entrada_flag else Decimal('0') npar = int(str(data.get('NÚMERO_PARCELAS') or '0') or 0) base = (total - v_ent) if (total - v_ent) > 0 else Decimal('0') parcela = (base / npar).quantize(Decimal("0.01"), rounding=ROUND_HALF_UP) if npar else Decimal('0.00') data['VALOR_PARCELAS'] = brl_fmt(parcela) if npar else '' ctx = normalize_context_keys(data) # Decodifica imagens imgs = decode_images_b64(images_b64) # Processa cada doc outputs = [] for name in docs: template_path = TEMPLATES_DIR / company / f"{name}.docx" if not template_path.exists(): return jsonify({"error": f"Template não encontrado: {template_path}"}), 400 # Gera DOCX out_docx = ensure_parent(BUILD_DIR / f"{company}_{name}_{datetime.now().strftime('%Y%m%d_%H%M%S')}.docx") tpl = DocxTemplate(str(template_path)) tpl.render(ctx) tpl.save(str(out_docx)) # Anexa imagens ao final if imgs: anexar_imagens_no_final(out_docx, imgs) # Converte para PDF nativo out_pdf = out_docx.with_suffix('.pdf') gerar_pdf(out_docx, out_pdf) outputs.append(out_pdf) # Resposta: PDF único ou ZIP if len(outputs) == 1: pdf = outputs[0] return send_file(str(pdf), as_attachment=True, download_name=pdf.name, mimetype='application/pdf') # ZIP múltiplo zip_buf = io.BytesIO() with zipfile.ZipFile(zip_buf, 'w', zipfile.ZIP_DEFLATED) as zf: for p in outputs: zf.write(p, arcname=p.name) zip_buf.seek(0) return send_file(zip_buf, as_attachment=True, download_name=f"{company}_documentos.zip", mimetype='application/zip') if __name__ == '__main__': app.run(debug=True) ``` --- ## templates/index.html (frontend — mantém Tailwind/JS, chama o backend Python) > É a **versão adaptada** do seu HTML: removi `docx-preview` + `html2pdf` e troquei a geração para `fetch('/api/generate')` enviando JSON (inclusive **imagens em base64**).
